\section{The Universal Type and P4 as Prohibition}
\label{sec:type}
% ══════════════════════════════════════════════════════

\subsection{RealProcess}

\begin{definition}[RealProcess]
A \emph{real process} is a function $R : \mathbb{N} \to \mathbb{Q}$.  We write $\RP := \natQ$.
\end{definition}

\begin{definition}[Cauchy condition]
A real process~$R$ is \emph{Cauchy} if for every $\varepsilon \in \mathbb{Q}$, $\varepsilon > 0$, there exists $N \in \mathbb{N}$ such that for all $m, n \ge N$:
\[
  |R(m) - R(n)| < \varepsilon.
\]
\end{definition}

In Rocq:
\begin{lstlisting}
Definition RealProcess := nat -> Q.

Definition is_Cauchy (R : RealProcess) : Prop :=
  forall eps : Q, 0 < eps ->
    exists N : nat, forall m n : nat,
      (N <= m)%nat -> (N <= n)%nat ->
      Qabs (R m - R n) < eps.
\end{lstlisting}

\begin{remark}
Under P4, the Cauchy process \emph{is} the real number.  There is no separate limit object---the process itself is the mathematical entity.
\end{remark}


\subsection{P4 as Prohibition}

The earlier paper presented P4 as a \emph{reinterpretation}: infinity can be replaced by processes.  We now establish the stronger result: P4 is \emph{inconsistent} with completed infinities.

\begin{definition}[Completed infinite set]
$\mathrm{CompletedInfSet}(S) :\Leftrightarrow \forall n : \mathbb{N},\; S(n)$.
\end{definition}

\begin{definition}[P4 stage-bounded actuality]
$\mathrm{P4\_bounded}(\mathrm{actual}) :\Leftrightarrow \forall \text{stage},\; \exists \text{bound},\; \forall n,\; \mathrm{actual}(\text{stage}, n) \Rightarrow n \le \text{bound}$.
\end{definition}

\begin{theorem}[P4 prohibits completed infinity]\label{thm:p4prohibit}
$\mathrm{CompletedInfSet}(S) \;\wedge\; \mathrm{P4\_bounded}(\mathrm{actual}) \;\wedge\; \mathrm{bridge}(S, \mathrm{actual}) \;\Rightarrow\; \bot.$
\end{theorem}

\begin{proof}
P4 gives a bound~$B$ for stage~0.  $\mathrm{CompletedInfSet}$ gives $S(B{+}1)$.  The bridge makes $B{+}1$ actual at stage~0.  But $B{+}1 > B$, contradicting the bound.
\end{proof}

\noindent
\emph{Rocq reference}: \texttt{P4CompletedInfinity.v}, theorem \texttt{completed\_inf\_contradicts\_P4}, \Qed.

\begin{keyresult}[Prohibition vs.\ Reinterpretation]
\emph{Reinterpretation}: P4 provides an \emph{alternative} to the Axiom of Infinity.\\
\emph{Prohibition}: P4 is \emph{inconsistent} with completed infinity.\\
Prohibition implies reinterpretation, but not vice versa.  We prove the stronger claim.
\end{keyresult}

Three consequences (\texttt{P4ProhibitsAC.v}, \texttt{P4ProhibitsImpredicative.v}):
\begin{enumerate}[nosep]
\item Full AC on~$\mathbb{N}$ produces a completed infinite object (the choice function) $\Rightarrow$ inconsistent with P4.
\item Impredicative definitions require quantifying over a completed totality $\Rightarrow$ blocked by P4 + P1.
\item Potential infinity (``always more, never all at once'') is \emph{compatible} with P4.
\end{enumerate}

Inductive Ord : Set :=

\end{lstlisting}

\noindent
Concrete ordinals:
\begin{align}
  \omega &= \texttt{OLim}(\texttt{nat\_to\_ord}), \\
  \omega^{\omega} &= \texttt{OLim}(\lambda n.\; \omega^n), \\
  \varepsilon_0 &= \texttt{OLim}(\texttt{omega\_tower}),
  \quad\text{where}\quad
  \texttt{omega\_tower}(0) = \omega,\;
  \texttt{omega\_tower}(n{+}1) = \omega^{\texttt{omega\_tower}(n)}.
\end{align}

Full arithmetic (addition, multiplication, exponentiation) is defined by structural recursion and verified.

\subsection{Well-Foundedness as Theorem}

\begin{theorem}[Well-foundedness of ordinal ordering]\label{thm:wf}
The relation $\texttt{ord\_lt}$ on $\texttt{Ord}$ is well-founded:
$\texttt{wf\_ord\_lt} : \texttt{well\_founded}\;\texttt{ord\_lt}$.
\end{theorem}

This is a \emph{theorem} (proved by structural induction on \texttt{Ord} with an accessibility argument), not an axiom.  The proof:

\begin{enumerate}[nosep]
\item \textbf{Helper}: $\texttt{acc\_succ}$---if $x$ is accessible, then $\texttt{OSucc}(x)$ is accessible (by induction on the \texttt{Acc} derivation).
\item \textbf{Main}: structural induction on \texttt{Ord}:
  \begin{itemize}[nosep]
  \item $\texttt{OZero}$: nothing is below zero (by inversion).
  \item $\texttt{OSucc}(a)$: use \texttt{acc\_succ} on the inductive hypothesis.
  \item $\texttt{OLim}(f)$: for each~$k$, $f(k)$ is accessible by IH; use \texttt{Acc\_inv}.
  \end{itemize}
\end{enumerate}

\noindent
\emph{Rocq reference}: \texttt{TransfiniteInduction.v}, \Qed.  No additional axioms.

\begin{corollary}[Transfinite induction]
For any property~$P$:
\[
  \bigl(\forall \alpha.\; (\forall \beta < \alpha.\; P(\beta)) \Rightarrow P(\alpha)\bigr) \;\Rightarrow\; \forall \alpha.\; P(\alpha).
\]
\end{corollary}

\subsection{ATR$_0$ Eliminated}

  match alpha with
  | OZero => base
  | OSucc a => P (iterate_pred P a base)
  | OLim f => fun n => forall k, iterate_pred P (f k) base n

\section{Classical Theorems as Process Constructions}
\label{sec:theorems}
% ══════════════════════════════════════════════════════

Every classical existence theorem in analysis has the same structure under~P4: construct an explicit process over~$\mathbb{Q}$, show it is Cauchy with a computable convergence rate, and verify it satisfies the required property at every stage.  We present eight theorems, each decomposed into the \ERR{} framework.

\subsection{Intermediate Value Theorem}

\begin{theorem}[IVT, Process Version]
Let $f : \mathbb{Q} \to \mathbb{Q}$ be uniformly continuous on $[a, b]$ with $f(a) < 0 < f(b)$.  Then there exists a Cauchy process $c : \mathbb{N} \to \mathbb{Q}$ with $c(n) \in [a, b]$ for all~$n$ and $f(c(n)) \to 0$.
\end{theorem}

\textbf{Process construction.}  Define $c(0) = (a + b)/2$.  At each step, the interval $[a_n, b_n]$ (with $f(a_n) < 0 < f(b_n)$) is bisected: if $f(c(n)) < 0$, set $a_{n+1} = c(n)$; otherwise $b_{n+1} = c(n)$.  The interval width halves:
\[
  |b_n - a_n| = \frac{b - a}{2^n}.
\]

\textbf{\ERR{} analysis.}
\begin{itemize}[nosep]
\item \textbf{Elements (L1):} midpoints $c(n)$ of nested intervals.
\item \textbf{Roles (L4):} each $c(n)$ is a rational approximation to the root; the process is Cauchy and bounded.
\item \textbf{Rules (L5):} bisection---evaluate~$f$ at the midpoint, choose the half where the sign change persists (using L3).
\end{itemize}

\noindent
\emph{Convergence rate}: $|c(m) - c(n)| \le (b-a)/2^{\min(m,n)}$ (geometric, $r = 1/2$).\\
\emph{Rocq}: \texttt{IVT\_ERR.v}, 23~\Qed.  Uses \texttt{classic} (L3) for the sign test.

\subsection{Extreme Value Theorem}

\begin{theorem}[EVT, Process Version]
Let $f : \mathbb{Q} \to \mathbb{Q}$ be uniformly continuous on $[a, b]$.  Then there exists a Cauchy process $x^* : \mathbb{N} \to \mathbb{Q}$ with $x^*(n) \in [a, b]$ such that $f(x^*(n))$ converges to $\sup_{x \in [a,b]} f(x)$.
\end{theorem}

\textbf{Process construction.}  At stage~$n$, evaluate~$f$ on a grid of $2^n + 1$ equally spaced points in $[a,b]$.  Set $x^*(n) = \arg\max_k f(a + k(b-a)/2^n)$.

\textbf{Key feature:} When multiple grid points achieve the maximum, L5-Resolution selects the \emph{leftmost}---not by convention but as the unique order-respecting choice (\texttt{L5Resolution.v}).

\textbf{\ERR{} analysis.}
\begin{itemize}[nosep]
\item \textbf{Elements:} grid maximizers $x^*(n)$ on a $2^n$-point grid.
\item \textbf{Roles:} each $x^*(n)$ is the best approximation on its grid; the supremum process $\{f(x^*(n))\}$ is monotonically non-decreasing.
\item \textbf{Rules:} grid refinement (doubling) + leftmost selection (L5).
\end{itemize}

\noindent
\emph{Convergence rate}: $\sup f - f(x^*(n)) \le \delta_f(2^{-n})$, where $\delta_f$ is the modulus of uniform continuity.\\
\emph{Rocq}: \texttt{EVT\_ERR.v}, 36~\Qed.

\subsection{Bolzano--Weierstrass Theorem}

\begin{theorem}[BW, Process Version]
Every bounded sequence $s : \mathbb{N} \to \mathbb{Q}$ in $[a, b]$ has a subsequential limit: there exists a nested interval process $[a_n, b_n]$ with $b_n - a_n = (b - a)/2^n$ such that infinitely many terms of~$s$ lie in $[a_n, b_n]$.
\end{theorem}

\textbf{Process construction.}  At each step, bisect the current interval.  At least one half contains infinitely many terms of~$s$ (by pigeonhole).  Choose that half, using L3 (\texttt{classic}) to decide which.

\textbf{\ERR{} analysis.}
\begin{itemize}[nosep]
\item \textbf{Elements:} nested intervals $[a_n, b_n]$.
\item \textbf{Roles:} each interval contains infinitely many terms of~$s$.
\item \textbf{Rules:} pigeonhole principle + L3 to choose the half.
\end{itemize}

\noindent
\emph{Convergence rate}: $b_n - a_n = (b-a)/2^n$ (geometric, $r = 1/2$).\\
\emph{Rocq}: \texttt{BolzanoWeierstrass.v}, 26~\Qed.

\subsection{Fundamental Theorem of Calculus}

\begin{theorem}[FTC, Process Version]
If $f : \mathbb{Q} \to \mathbb{Q}$ is Lipschitz on $[a, b]$ with antiderivative process $F(n) = \sum_{k=0}^{2^n-1} f(x_k) \cdot \Delta x_n$ (Riemann sums on a $2^n$-point grid), then $\{F(n)\}$ is Cauchy and $F'(x) = f(x)$ in the process sense.
\end{theorem}

\textbf{\ERR{} analysis.}
\begin{itemize}[nosep]
\item \textbf{Elements:} Riemann sums $F(n)$.
\item \textbf{Roles:} each $F(n)$ approximates the integral; the process is Cauchy.
\item \textbf{Rules:} grid refinement; Lipschitz condition bounds the error:
\[
  |F(n+1) - F(n)| \le L \cdot (b - a)^2 / 2^{n+1},
\]
where $L$ is the Lipschitz constant.
\end{itemize}

\noindent
\emph{Rocq}: \texttt{analysis/FTC.v}, 28~\Qed.

\subsection{Mean Value Theorem}

\begin{theorem}[MVT, Process Version]
If $f$ is differentiable on $(a,b)$ and continuous on $[a,b]$, then there exists a process $c : \mathbb{N} \to \mathbb{Q}$ with $c(n) \in (a,b)$ such that
\[
  f'(c(n)) \to \frac{f(b) - f(a)}{b - a}.
\]
\end{theorem}

The process is constructed by applying IVT to $g(x) = f(x) - x \cdot \frac{f(b)-f(a)}{b-a}$, which satisfies $g(a) = g(b)$, then finding the extremum of~$g$ via EVT.

\emph{Rocq}: \texttt{MeanValueTheorem.v}, 18~\Qed.

\subsection{Taylor's Theorem}

\begin{theorem}[Taylor, Process Version]
If $f$ has $n$ derivatives on $[a,b]$, then the Taylor partial sum process
\[
  T_n(x) = \sum_{k=0}^{n} \frac{f^{(k)}(a)}{k!}(x - a)^k
\]
satisfies $|f(x) - T_n(x)| \le M |x-a|^{n+1}/(n+1)!$, where $M$ bounds $|f^{(n+1)}|$.
\end{theorem}

\emph{Rocq}: \texttt{TaylorSeries.v}, 18~\Qed.  The bound is verified as a rational inequality.

\subsection{Picard--Lindel\"of Theorem}

\begin{theorem}[Picard, Process Version]
If $F : \mathbb{Q}^2 \to \mathbb{Q}$ is Lipschitz in the second argument with constant~$L$, and $y_0 \in \mathbb{Q}$, then the Picard iterates
\[
  y_{n+1}(t) = y_0 + \int_0^t F(s, y_n(s))\, ds
\]
form a Cauchy process with convergence rate $(Lt)^n / n!$.
\end{theorem}

\textbf{\ERR{} analysis.}
\begin{itemize}[nosep]
\item \textbf{Elements:} Picard iterates $y_n(t)$.
\item \textbf{Roles:} each $y_n$ is a better approximation to the ODE solution.
\item \textbf{Rules:} Lipschitz contraction; the error satisfies $\|y_{n+1} - y_n\| \le \frac{(Lt)^n}{n!} \|y_1 - y_0\|$.
\end{itemize}

\emph{Rocq}: \texttt{PicardLindelof.v}, 23~\Qed.

\section{Process Completeness}
\label{sec:completeness}

A standard objection: ``Without~$\mathbb{R}$, you lose completeness---Cauchy sequences may not converge.''  We answer: Cauchy processes \emph{do} converge, to a \emph{process}---not to a point in a completed set.

\subsection{Metric on RealProcess}

We define a stagewise distance between processes:

\begin{definition}[Stage distance]
For processes $R, S : \RP$ and truncation level~$N$:
\[
  d_N(R, S) = \sum_{k=0}^{N-1} |R(k) - S(k)|.
\]
\end{definition}

\noindent
Properties (all verified in Rocq):
\begin{align}
  d_N(R, R) &= 0, \label{eq:self}\\
  d_N(R, S) &= d_N(S, R), \label{eq:sym}\\
  d_N(R, S) &\le d_{N+1}(R, S). \label{eq:mono}
\end{align}

Under P4, the ``full'' distance $d(R, S) = \lim_{N \to \infty} d_N(R, S)$ does not exist as a completed object.  The sequence $\{d_N(R, S)\}_{N \in \mathbb{N}}$ is itself a process over~$\mathbb{Q}$.

\emph{Rocq}: \texttt{ProcessMetricComplete.v}, theorems \texttt{proc\_dist\_N\_self}, \texttt{proc\_dist\_N\_sym}, \texttt{proc\_dist\_mono}.

\subsection{Diagonal Limit Construction}

\begin{definition}[Process-Cauchy]
A sequence of processes $\mathrm{seq} : \mathbb{N} \to \RP$ is \emph{process-Cauchy} if for every truncation~$N$ and every $\varepsilon > 0$, there exists~$M$ such that for all $i, j \ge M$:
\[
  d_N(\mathrm{seq}(i), \mathrm{seq}(j)) < \varepsilon.
\]
\end{definition}

\begin{definition}[Diagonal process]
Given $\mathrm{seq} : \mathbb{N} \to \RP$, the \emph{diagonal process} is:
\[

\]
\end{definition}

The diagonal takes the $k$-th value from the $k$-th process---always a concrete rational.

\begin{theorem}[Stagewise completeness]\label{thm:completeness}
If $\mathrm{seq}$ is process-Cauchy, then for every stage~$k$ and every $\varepsilon > 0$, there exists~$M$ such that for all $i \ge M$:
\[
  |\mathrm{seq}(i)(k) - R_\infty(k)| < \varepsilon.
\]
\end{theorem}

\begin{proof}
Process-Cauchy at truncation $N = k{+}1$ gives $M_k$ such that $|\mathrm{seq}(i)(k) - \mathrm{seq}(j)(k)| < \varepsilon$ for $i, j \ge M_k$.  Take $M = \max(M_k, k)$ and $j = k$:
$|\mathrm{seq}(i)(k) - \mathrm{seq}(k)(k)| = |\mathrm{seq}(i)(k) - R_\infty(k)| < \varepsilon$.
\end{proof}

\section{Fourier Analysis Over $\mathbb{Q}$}

Classical Fourier analysis requires $L^2(\mathbb{R})$---a completed Hilbert space.  Under P4, we replace it with finite DFT on graphs, where everything is exact rational arithmetic.

\subsection{DFT on Cycle Graph}

The cycle graph~$C_N$ has adjacency eigenvalues $\lambda_k = 2\cos(2\pi k/N)$ for $k = 0, \ldots, N{-}1$.  The DFT decomposes a signal $f : \{0, \ldots, N{-}1\} \to \mathbb{Q}$ into eigenmodes:
\[
  \hat{f}_k = \frac{\langle f, \varphi_k \rangle}{\|\varphi_k\|^2},
  \quad\text{where } \varphi_k \text{ is the } k\text{-th eigenvector of } C_N.
\]

For $N = 4$: eigenvalues $\{2, 0, -2, 0\}$, eigenvectors $\varphi_0 = (1,1,1,1)$, $\varphi_1 = (1,0,-1,0)$, $\varphi_2 = (1,-1,1,-1)$, $\varphi_3 = (0,1,0,-1)$.  All entries rational.  All inner products exact.

\begin{theorem}[Parseval over $\mathbb{Q}$]

This is verified by \texttt{vm\_compute} for $N = 2$ (general signal, symbolic in $a, b$) and for $N = 4$ (five concrete test signals).  No floating-point---exact $\mathbb{Q}$-arithmetic.

\subsection{PMG Criterion}

\begin{definition}[Process Mass Gap]
A process $g : \mathbb{N} \to \mathbb{Q}$ has process mass gap~$\varepsilon$ if:
\begin{enumerate}[nosep]
\item[PMG1.] $g(M) \ge \varepsilon > 0$ for all $M \ge 0$ (uniform lower bound).
\item[PMG2.] $|g(M{+}1) - g(M)| \le C r^M$ for some $C > 0$, $0 < r < 1$ (Cauchy rate).
\item[PMG3.] $g(M{+}1) \ge g(M)$ for all $M$ (monotonicity).
\end{enumerate}
\end{definition}

PMG is strictly stronger than the standard $\lim g(M) > 0$: a process that is zero at one scale (e.g., $g(5) = 0$) fails PMG1 but satisfies the standard criterion.  Under P4, the process \emph{is} the physics at every scale.
